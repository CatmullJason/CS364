    <th>Method Pseudo Code</th>
  </tr>
  <tr>
    <td>assignment.setCompleted(false)</br>
           if isTask(assignment):</br>
	   taskList.add(assignment)</br>
           if isEvent(assignment):</br>
           eventList.add(assignment)
